STL GEOMETRY CHECK

import trimesh

def check_stl(file_path):
    mesh = trimesh.load(file_path)
    
    # 1. Check for Holes (Watertight)
    is_watertight = mesh.is_watertight
    
    # 2. Check Resolution
    face_count = len(mesh.faces)
    
    # 3. Decision
    if not is_watertight:
        return {"status": "FAIL", "reason": "Mesh has holes"}
        
    if face_count < 10000:
        return {"status": "FAIL", "reason": "Low resolution"}
        
    return {"status": "PASS"}

def validate_release(udi, docs):
    if not docs.get("dhr"):
        return "FAIL: Missing DHR"

    if docs.get("sterilization_status") != "PASS":
        return "FAIL: Sterilization not passed"

    if days_since(docs.get("calibration_date")) > 180:
        return "WARN: Calibration overdue"

    if docs.get("udi_traceability") != "OK":
        return "WARN: UDI chain incomplete"

    return "PASS"
