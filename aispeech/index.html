# 异动归因 Python 代码示例
def detect_anomaly(df, metric, threshold=0.15):
    df['wow'] = df[metric].pct_change(7)  # 周同比
    anomalies = df[abs(df['wow']) > threshold]
    return anomalies

def drill_down(anomaly, dimensions):
    """按维度下钻找根因"""
    results = []
    for dim in dimensions:
        contribution = calc_contribution(anomaly, dim)
        if contribution > 0.3:  # 贡献度 > 30%
            results.append({'dim': dim, 'contrib': contribution})
    return sorted(results, key=lambda x: -x['contrib'])

# Bug 聚类与评级示例
def cluster_bugs(new_bug, existing_bugs, threshold=0.85):
    """基于语义相似度聚类"""
    embedding = get_embedding(new_bug.description)
    for bug in existing_bugs:
        similarity = cosine_similarity(embedding, bug.embedding)
        if similarity > threshold:
            return {'action': 'merge', 'target': bug.id}
    return {'action': 'create'}

def rate_severity(bug):
    """基于规则 + LLM 判定严重级别"""
    rules_score = apply_rules(bug)  # 关键词/模块规则
    llm_score = llm_assess(bug)      # LLM 辅助判断
    return weighted_avg(rules_score, llm_score, weights=[0.6, 0.4])

06 · 风险

主要风险与缓解策略

数据安全

私有化/专有云部署，敏感字段脱敏，按角色授权，审计日志保留。

口径漂移

知识库版本管理 + 评测集验证，变更需通过自动化评测。

模型稳定性

缓存/多模型回退策略，关键链路加规则校验兜底。

采纳度

模板市场 + 培训机制，设置 Owner 与 KPI 绑定。

06 · FAQ

常见疑问

  • 能否复用现有系统？ 支持 API/Webhook 对接 JIRA、IM、语音转写、BI 等。
  • 合规如何做？ 权限分级、审计日志、脱敏策略、模型黑白名单。
  • 如何控制成本？ 调用限额、缓存分层、离线批处理替代实时。
  • 谁来维护？ 每条 Workflow 绑定 Owner + 指标，运营节奏周迭代。
  • 效果不好怎么办？ 评测集驱动，A/B 实验验证，持续优化 Prompt/知识库。
